-- Soal 1.a Membuat trigger tambahStok dengan kondisi setelah penginputan di table suplai_produk otomatis bertambah ke trigger attribute stok di table produk 

DELIMITER $$
CREATE TRIGGER tambahStok
AFTER INSERT ON suplai_produk
FOR EACH ROW
BEGIN
	UPDATE produk SET stok = stok + NEW.jumlah WHERE id = NEW.produk_id;
END$$
DELIMITER ;
SELECT * FROM produk;

-- Soal 1.b Membuat trigger normalisasiStok dgn kondisi transaksi penginputan suplai_produk dibatalkan (delete), maka akan mentrigger stok kembali normal di tabel produk

DELIMITER $$
CREATE TRIGGER normalisasiStok
AFTER DELETE ON suplai_produk
FOR EACH ROW
BEGIN
	UPDATE produk SET stok = stok - old.jumlah WHERE id = old.produk_id;
END$$
DELIMITER ;
SELECT * FROM produk;

-- Soal 2 query inner join ( produk + jenis + suplai_produk + suplier) Tampilkan data tgl_suplai, nama produk, nama jenis, jumlah barang yang disuplai, nama supplier (relasi 4 buah tabel)

-- use db toko_online
USE toko_online;    

SELECT suplai_produk.tgl as tgl_suplai,
    produk.nama as nama_produk,
    jenis.nama as nama_jenis,
    suplai_produk.jumlah as jml_suplai,
    suplier.nama as nama_suplier
FROM suplai_produk
	INNER JOIN produk on suplai_produk.produk_id = produk.id
    INNER JOIN jenis on produk.jenis_id = jenis.id
    INNER JOIN suplier on suplai_produk.suplier_id = suplier.id;
